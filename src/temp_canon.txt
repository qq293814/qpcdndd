InitNewCanon_Function: function() {
    var self = this;
    var canonBody = "Texture/Canon/fish_cannon_19_body.png";
    cc.loader.loadRes(canonBody, cc.SpriteFrame, function(err, sf) {
        if (!err && sf && self.pt_PlayerCanon) {
            for (var i = 0; i < self.pt_PlayerCanon.length; i++) {
                var node = self.pt_PlayerCanon[i];
                if (node) {
                    var sp = node.getComponent(cc.Sprite);
                    if (sp) sp.spriteFrame = sf;
                }
            }
        }
    });
},
